        正则表达式搜索：使用search()方法来匹配
    t_list=bs.find_all(re.compile("a"))

    用方法来搜索：传入一个函数，根据函数的要求来搜索
    def name_is_exists(tag):
        return tag.has_attr("name")#搜的tag标签里面要有name属性，也就是用函数缩小搜索范围。
    t_list=bs.find_all(name_is_exists)
    for item in t_list:
        print(item)
