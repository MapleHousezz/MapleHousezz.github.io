爬虫第1课：发送与读取请求

2-26-21
    import urllib.request,urllib.error 导入库
    获取一个get请求
    response=urllib.request.urlopen("http://www.baidu.com")
    print(response.read().decode("utf-8"))#封装之后读取，再解码。

    获取一个post请求,需要提交数据表单
    import urllib.parse
    data = bytes(urllib.parse.urlencode({"hello":"world","1":"2"})encoding="utf-8")#这里是处理附加的表单信息
    response2=urllib.request.urlopen("http://httpbin.org/post",data=data) 
    print(response2.read().decode("utf-8"))

    超时处理
    try:
            response=urllib.request.urlopen("http://httpbin.org/get",timeout=0.01)
            print(response.read().decode("utf-8"))
    except urllib.error.URLError as e:
            print("time out")

    response=urllib.request.urlopen("http://www.baidu.com")
    print(response.status)#状态码，418是被发现了
    print(response.getheaders())#响应头getheaders 要加s除非爬取单个
    print(response.getheader("Server"))#提取相应键值对

    url="http://httpbin.org/post"
    headers={
                "User-Agent": "Mozilla/6.0 (Windows NT 10.0; Win64; x64; rv:86.0) Gecko/20100101 Firefox/86.0"
                }
    data=bytes(urllib.parse.urlencode({"name":"eric"}),encoding="utf-8")
    req=urllib.request.Request(url=url,headers=headers,data=data,method="POST")
    response=urllib.request.urlopen(req)
    print(response.read().decode("utf-8"))
